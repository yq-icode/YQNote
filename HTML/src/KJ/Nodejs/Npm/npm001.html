常用命令

安装与卸载

	            	
//安装模块的特定版本
npm install sax@0.1.1     //本地安装
npm install sax@0.1.1 -g  //全局安装

//卸载
npm uninstall sax@0.1.1

//查看安装路径
where npm     //查看 npm 安装路径
where node    //查看 node 安装路径
where webpack //查看 webpack 安装路径

//生成package.json文件
npm init

//为项目添加webpack依赖
npm install webpack --save-dev

//指定配置文件
webpack --config webpack.dev.config.js

查看版本

	            	
//查看本地已安装模块清单
npm list

//查看本地已安装模块版本
npm list packageName

//查看某模块的详细信息 包括各版本号等
npm info packageName

//查看某模块远程最新版本
npm view packageName version

//查看某模块远程所有版本
npm view packageName versions

//查看某模块版本号
webpack -v //查看webpack版本号
