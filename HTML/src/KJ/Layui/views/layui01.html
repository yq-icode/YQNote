修改 option 颜色

option标签中只能放文本内容，即使放了其它标签，也会被剔除，即在 debug-审查元素中，会发现标签被自动删除掉了。因此只能将 标签+文本的组合 以字符串形式放在 option的自定义属性中

	                   
<select name="KCName" lay-verify="required">
	<option value="">请选择</option>
	<option value="春雨科目三aaa考场" data-str="春雨科目三<label style='color: red;'>aaa</label>考场">春雨科目三aaa考场</option>
	<option value="河南小兴考场">河南小兴考场</option>
</select>
		            
	            
	                   
if($('select').length > 0){
	$('select').each(function(){
		var D_Select = $(this);
		$(D_Select).find('option').each(function(){
			var D_Option = $(this);
			if($(D_Option).attr('data-str')){
				var O_Idx = $(D_Option).index(),
					O_DStr = $(D_Option).attr('data-str');
				$(D_Select).next('.layui-form-select').find('dl dd').eq(O_Idx).html(O_DStr);
			}
		})
	})
}
