控制元素的隐藏和显示

	                   
<div id="box">
    <input type="button" value="toggle" v-on:click="toggle()">
    <div v-show="isShow">点击上面的按钮可控制此DIV显示与隐藏</div>
</div>
		            
	            
	                   
var vm = new Vue({
    el:'#box',
    data:{
        isShow:false
    },
    methods:{
        toggle:function(){
            this.isShow = !this.isShow;
        }
    }
});
