var video = document.querySelector('#video');

基本操作

播放、暂停

                
//paused设置或返回音频/视频是否暂停                	
if(video.paused){
	video.play();
}
else{
	video.pause();
}

指定播放时间

currentTime设置或返回音频/视频中的当前播放位置（以秒计）

                
video.currentTime = 20
video.currentTime = video.duration //结束视频（duration返回当前音频/视频的长度（以秒计））
