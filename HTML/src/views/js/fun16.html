}

前一天、后一天

	                
//前一天                	
datePre:function(){
	let d = new Date('2017-10-10');
	d.setDate(d.getDate() - 1);
	let _d = dateToString(d);
}
//后一天
dateNext:function(){
	let d = new Date('2017-10-10');
	d.setDate(d.getDate() + 1);
	let _d = dateToString(d);
}
dateToString:function(d){
	let year = d.getFullYear(),
		month = d.getMonth() + 1,
		day = d.getDate();
	month = month > 9 ? month : '0' + month;
	day = day > 9 ? day : '0' + day;
	let date = year + "." + month + "." + day;
	return date;
}
