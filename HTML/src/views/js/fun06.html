jQuery.inArray

确定第一个参数在数组中的位置，从0开始计数(如果没有找到则返回 -1 )。

value
用于在数组中查找是否存在
array
待处理数组
fromIndex
用来搜索数组队列，默认值为0
                
jQuery.inArray(value,array,[fromIndex])
$.inArray(value,array,[fromIndex])

var arr = [ 4, "Pete", 8, "John" ];

jQuery.inArray("John", arr);  //3
jQuery.inArray(4, arr);  //0
$.inArray("David", arr);  //-1
$.inArray("Pete", arr, 2);  //-1
