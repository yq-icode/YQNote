常用验证

phone

手机号规则：第一位是【1】开头，第二位则则有【3,4,5,7,8】，第三位则是【0-9】，第三位之后则是数字【0-9】

					
                        function checkPhone(phoneNumber){ 
							if(!(/^1[34578]\d{9}$/.test(phoneNumber))){ 
								alert("手机号码有误，请重填");  
								return false; 
							}else{
								return true;
							}
						}

身份证号

身份证号规则：身份证号码为15位或者18位，15位时全为数字，18位前17位为数字，最后一位是校验位，可能为数字或字符X|x

					
                        function checkSFZ(sfz){ 
							var regIdNo = /(^\d{15}$)|(^\d{18}$)|(^\d{17}(\d|X|x)$)/; 
							if(!regIdNo.test(sfz)){ 
								alert(‘身份证号填写有误‘); 
								return false; 
							}else{
								return true;
							}
						}
