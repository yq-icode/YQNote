关于对象数据的操作

为对象追加新数据

		                
//法一		                	
let obj = {"name":"刘丽", "age":22};
let key = "sex", value = "女";
obj[key] = value;
console.log(obj); //{age:22, name:"刘丽", sex:"女"}

//法二
var obj1 = {"name":"刘丽", "age":22};
var obj2 = {"sex":"女"};
var obj3 = {
	...obj1,
	...obj2
}
console.log(obj3); //{age:22, name:"刘丽", sex:"女"}

//复制对象
var obj1 = {"name":"刘丽"};
var obj2 = {"age":22};
var obj3 = Object.assign({}, obj1, obj2);
console.log(obj3); //{age:22, name:"刘丽"}

//合并对象
var obj1 = {"name":"刘丽"};
var obj2 = {"age":22};
var obj3 = {"sex":"女"};
Object.assign(obj1, obj2, obj3);
console.log(obj1); //{age:22, name:"刘丽", sex:"女"}

//合并具有相同属性的对象
var obj1 = {"name":"刘丽", "age":22};
var obj2 = {"age":28};
var obj3 = {"sex":"女"};
Object.assign(obj1, obj2, obj3);
console.log(obj1); //{age:28, name:"刘丽", sex:"女"}
