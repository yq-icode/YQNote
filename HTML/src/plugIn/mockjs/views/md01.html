模拟日期与时间

模拟指定范围的日期与时间段

  • 模拟日期范围：10天前 至 当天
  • 模拟时间范围：时间段不能跨天（即不能显示成 23:00 - 01:00）
  • 排序：日期按时间先后排序
md01.js

phone

					
var Random = Mock.Random;
Random.extend({
	phone: function(){
		var phonePrefixs = ['132', '135', '189']; //手机号前缀
		return this.pick(phonePrefixs) + Mock.mock(/\d{8}/);
	}
})

//调用方法
"student":{
	"phone": "@phone"
}
